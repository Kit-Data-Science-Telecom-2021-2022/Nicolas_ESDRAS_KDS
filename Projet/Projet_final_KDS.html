VMG_rang_24h = pd.DataFrame(df_class.groupby('Rang')['VMG24 (kts)'].mean())
Rang = VMG_rang_24h.index.values.reshape(-1,1)
VMG_24h = VMG_rang_24h['VMG24 (kts)']

reg = LinearRegression(fit_intercept=True)
reg.fit(Rang, VMG_24h)
VMG_pred = reg.predict(Rang)

coef = np.round(reg.coef_[0],4)
intercept = np.round(reg.intercept_,4)
r2 = np.round(sklearn.metrics.r2_score(VMG_24h, VMG_pred),4)

print(f'Correlation entre le rang et la vitesse VMG depuis 24 heures:\n\n\
- Coefficient : {coef}\n\
- Intercept : {intercept}\n\
- Score R2 : {r2}\n')
